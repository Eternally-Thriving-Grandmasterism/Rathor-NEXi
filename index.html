// ──────────────────────────────────────────────────────────────── // Imports import { MeTTaRewriter } from '/metta-rewriting-engine.js'; import { atomeseValenceGate } from '/atomese-knowledge-bridge.js'; import { hyperonValenceGate } from '/hyperon-reasoning-layer.js'; // IndexedDB setup let db; const dbName = "rathorChatDB"; const storeName = "messages"; const dbRequest = indexedDB.open(dbName, 1); dbRequest.onupgradeneeded = event => { db = event.target.result; db.createObjectStore(storeName, { keyPath: "id", autoIncrement: true }); }; dbRequest.onsuccess = event => { db = event.target.result; loadChatHistory(); }; function loadChatHistory() { const tx = db.transaction(storeName, "readonly"); const store = tx.objectStore(storeName); const req = store.getAll(); req.onsuccess = () => { req.result.forEach(msg => addChatMessage(msg.role, msg.content)); }; } function saveMessage(role, content) { const tx = db.transaction(storeName, "readwrite"); const store = tx.objectStore(storeName); store.add({ role, content, timestamp: Date.now() }); } function clearChatHistory() { const tx = db.transaction(storeName, "readwrite"); const store = tx.objectStore(storeName); store.clear(); document.getElementById('chat-messages').innerHTML = ''; } function addChatMessage(role, content) { const div = document.createElement('div'); div.className = `chat-message ${role}`; div.textContent = content; document.getElementById('chat-messages').appendChild(div); document.getElementById('chat-messages').scrollTop = document.getElementById('chat-messages').scrollHeight; saveMessage(role, content); } // Multi-layer valence gate – MeTTa first, then Atomese + Hypergraph async function multiLayerValenceGate(text) { // Stage 1: MeTTa rewriting (first mercy gate) const mettaResult = MeTTaRewriter.rewrite(text); // Early rejection if rewrite valence too low if (mettaResult.rejected) { return { result: 'REJECTED', valence: mettaResult.valenceImpact, reason: mettaResult.reason }; } // Stage 2: Atomese grounding + valence const atomeseResult = await atomeseValenceGate(mettaResult.rewritten || text); // Stage 3: Hyperon hypergraph reasoning const hyperonResult = await hyperonValenceGate(mettaResult.rewritten || text); // Weighted fusion: MeTTa 35%, Atomese 30%, Hyperon 35% const finalValence = (mettaResult.valenceImpact + 1) * 0.35 + // normalize (-1..+1 → 0..2) atomeseResult.valence * 0.30 + hyperonResult.valence * 0.35; const finalReason = [ mettaResult.rewritten !== text ? `MeTTa rewritten: ${mettaResult.rewritten}` : '', atomeseResult.reason, hyperonResult.reason ].filter(r => r).join(' | ') + ` → final valence ${finalValence.toFixed(7)}`; return { result: finalValence >= 0.9999999 ? 'ACCEPTED' : 'REJECTED', valence: finalValence, reason: finalReason }; } // Chat logic with full MeTTa → Atomese → Hypergraph pipeline async function sendChat() { const text = document.getElementById('chat-input').value.trim(); if (!text) return; const gateResult = await multiLayerValenceGate(text); if (gateResult.result === 'REJECTED') { addChatMessage('rathor', `Multi-layer valence gate held: ${gateResult.reason}`); document.getElementById('chat-input').value = ''; return; } addChatMessage('user', text); document.getElementById('chat-input').value = ''; const reply = `NEXi + MeTTa + Atomese + Hypergraph approved: "${text}". Truth echoes.`; addChatMessage('rathor', reply); } document.getElementById('chat-input').addEventListener('keydown', e => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendChat(); } }); document.getElementById('clear-history').addEventListener('click', clearChatHistory); if ('serviceWorker' in navigator) { navigator.serviceWorker.register('/sw.js') .then(reg => console.log('SW registered')) .catch(err => console.log('SW failed', err)); }
